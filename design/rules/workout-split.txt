# Determining the split

IF frequency is 3 days 
THEN 
    IF user preference is Full body 
    THEN Workout split is Full body
    ELSE Workout split is Push-pull-leg
ELSE 
    IF frequency is 4 days
    THEN Workout split is Upper-Lower
ELSE 
    IF frequency is 6 days 
    THEN Workout split is Push-pull-leg

IF goal is Strength
    Repetition range is 1-5
ELSE IF goal is Hypertrophy
    Repetition range is 8-12
ELSE IF goal is Endurance
    Repetition range is 15+

IF exercise does not have a set count assigned THEN
    IF muscle group priority is high THEN
        Number of set is 3
    ELSE 
        Number of set is 2

# Logic to assign order
FOR ALL days
    IF an exercise-slot previous-slot with order o-highest AND
        the order is the highest among exercise-slots AND
        an exercise-slot unassigned-slot with no order AND
        unassigned-slot does not targets the same muscle group as previous-slot AND
        unassigned-slot has the highest priority among the unordered slots AND
    THEN 
        unassigned-slot is assigned the next order

# Start the day with the user weak muscle group
FOR ALL days 
    IF the day contains an exercise slot AND
        the slot targets the user weak muscle group
    THEN 
        1he slot has order 1
